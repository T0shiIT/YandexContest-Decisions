**2. Список студентов.**
```cpp
#include <iostream> // Подключаем стандартную библиотеку ввода‑вывода C++.
#include <vector> // Подключаем библиотеку для работы с динамическими массивами (std::vector).
#include <string> // Подключаем библиотеку для работы со строками (std::string).
#include <algorithm> // Подключаем библиотеку с готовыми алгоритмами

struct Student { // Создаём структуру (тип данных) Student.
    std::string name; // name — имя (строка),
    std::string group; // group — группа (строка).
};

using namespace std; // Используем пространства имён.

Student make_student(string line) { // Функция, которая создаёт объект Student из строки line. Возвращает структуру Student.
    size_t pos = line.find(';'); // Находим позицию символа ; в строке line.
								 // size_t — специальный тип для хранения индексов/размеров.
								 // find(';') возвращает номер символа ; (или std::string::npos, если не найден).
	
    string name = line.substr(0, pos); // Выделяем подстроку от начала (0) до позиции pos (не включая pos). Это будет имя студента.
    string group = line.substr(pos + 1, line.length() - pos - 1); // Выделяем подстроку после ; (pos + 1).
																  // Длина подстроки: line.length() - pos - 1 (весь остаток строки).
																  // Это будет группа студента.
	
    return {name, group}; // Возвращаем структуру Student, заполненную name и group.
}

bool is_upper(Student a, Student b) { // Функция-компаратор для std::sort. Возвращает true, если a должен идти раньше b в отсортированном списке.
									  // Она определяет порядок сортировки студентов в списке. Её задача — ответить на вопрос:
									  // «Должен ли студент a идти в списке раньше студента b?»: Если да — возвращает true, если нет — false.

    if (a.group == b.group) { // Если группы студентов совпадают, сортируем по имени.
        return a.name < b.name; // Возвращаем true, если имя a лексикографически меньше имени b.
    } else { // Если группы разные, сортируем по названию группы.
        return a.group < b.group; // Возвращаем true, если группа a меньше группы b.
    }
}

void print(vector<Student> students) { // Функция для вывода списка студентов. Принимает вектор students.
    if (students.empty()) { // Если вектор пуст (empty()).
        cout << "Empty list!"; // Выводим «Empty list». 
    } else { // Иначе (если студенты есть):
        for (size_t i = 0; i < students.size(); i++) { // Цикл по всем студентам (i от 0 до students.size() - 1).
            if (i == 0 || students[i].group != students[i-1].group) { // Если это первый студент (i == 0) или его группа отличается от предыдущего (students[i-1]).
                cout << students[i].group << endl; // Выводим название группы.
            }
            cout << "- " << students[i].name << endl; // Выводим имя студента с префиксом - .
        }
    }
}

int main() // Объявляем главную функцию программы, с которой начинается выполнение.
{
    int count; // Количество студентов.
    std::cin >> count; // Считываем число count (количество студентов) с клавиатуры.
    std::cin.ignore(1);  // Убираем из потока символ \n для корректной работы getline
    
    std::vector<Student> students(count); // Создаём вектор students размером count (пустой, но зарезервированный).
    for(auto& student: students) { // Цикл по всем элементам вектора students. 
								   // auto& student — ссылка на текущий элемент (чтобы не копировать).
	
        std::string line; // Cоздаем переменную типа string.
        std::getline(std::cin, line); // Считываем всю строку (включая пробелы) в line.
        student = make_student(line); // Преобразуем строку line в структуру Student и сохраняет в student.
    }
    
    std::sort(students.begin(), students.end(), is_upper); // Сортируем вектор students:
														   // students.begin() — начало вектора,
														   // students.end() — конец вектора,
														   // is_upper — функция-компаратор (определена выше).
    
    print(students); // Выводим отсортированный список студентов.
}
```

**3. Кандидаты на отчисление.**
```cpp
#include <iostream> // Подключаем стандартную библиотеку ввода‑вывода C++.
#include <algorithm> // Подключаем библиотеку, в которой лежат готовые функции для работы с данными: сортировка, поиск, копирование и т.д
#include <iterator> // Подключаем библиотеку итераторов — специальных “указателей” на элементы в контейнерах (векторах, списках и т.п.).
#include <string> // Подключаем библиотеку для работы со строками (std::string).
#include <vector> // Подключаем библиотеку для работы с динамическими массивами (std::vector).

struct Course { // Создаём структуру (тип данных) под названием Course (курс).
    std::string name; // Название курса (например, “Матанализ”).
    int semester; // Номер семестра, когда курс должен был быть пройден.
    bool finished; // Закончен ли курс (да/нет, true/false).
};

struct Student { // Создаём структуру Student (студент).
    std::string name; // Имя студента.
    std::string group; // Номер группы (например, “ИВТ-123”).
    std::vector<Course> courses; // Вектор (список) всех курсов, которые у него были.
};

using namespace std; // Используем пространства имён.

Student make_student(string line) { // Объявляем функцию make_student, которая принимает строку line и возвращает объект типа Student.
    string name, group; // Объявляем переменные, которые будут заполнены из строки:
						// name — имя студента.
						// group — номер группы.

    vector<Course> courses; // Список его курсов.
    size_t start = 0; // Позиция, с которой начинается текущая “часть” строки (например, имя).
    size_t pos = line.find(';'); // Позиция первого символа ; в строке. Функция find(';') ищет точку с запятой.
    vector<string> stud; // Создаём вектор строк stud, в который будут складываться все части строки, разделённые ;.
    while (pos != string::npos) { // Цикл продолжается, пока pos не равен string::npos — это специальное значение, которое означает “ничего не найдено”.
        stud.push_back(line.substr(start, pos - start)); // push_back — добавляет её в вектор stud.
														 // substr(start, pos - start) — вырезает подстроку от start до pos (не включая ;).

        start = pos + 1; // Сдвигаем за пределы текущей точки с запятой (на следующий символ).
        pos = line.find(';', start); // Ищем следующую точку с запятой, начиная с start.
    }
    stud.push_back(line.substr(start)); // После цикла добавляется последняя часть строки — от последнего ; до конца.
    name = stud[0]; // Первый элемент (stud[0]) — имя студента.
    group = stud[1]; // Второй (stud[1]) — группа.
    for (size_t i = 2; i < stud.size(); i += 3) { // Цикл начинается с индекса 2 (т.к. 0 и 1 — имя и группа), и шаг — 3, потому что каждый курс описывается тремя полями: название, семестр, флаг завершения.
        Course course = {stud[i], stoi(stud[i + 1]), stud[i + 2] != "0"}; // Создаём объект Course:
																		  // stud[i] — название курса (строка).
																		  // stoi(stud[i + 1]) — превращает строку в число (например, "2" → 2).
																		  // stud[i + 2] != "0" — если строка не "0", значит курс закончен (true), иначе false.
		
        courses.push_back(course); // Добавляем созданный курс в список курсов студента.
    }
    return {name, group, courses}; // Возвращаем структуру Student, созданную из name, group, courses.
}

bool is_upper(Student a, Student b) { // Функция сравнения для сортировки. Нужна для std::sort.
    if (a.group == b.group) { // Если группы одинаковые — сортируем по имени (алфавиту).
        return a.name < b.name; 
    } else { // Иначе — по номеру группы.
        return a.group < b.group;
    }
}

bool is_debtor(Student student, int cur_semester, int max_debt) { // Функция проверяет, является ли студент “должником” (т.е. имеет слишком много несданных курсов).
    int debts = 0; // Счётчик долгов (несданных курсов).
    for (Course c: student.courses) { // Цикл по всем курсам студента (c — текущий курс).
        if (c.semester < cur_semester) { // Если курс был в прошлых семестрах (c.semester < cur_semester),
            if (!c.finished) { // И он не закончен (c.finished),
                debts++; // То считаем его долгом.
            }
        }
    }
    return (debts > max_debt) ? true : false; // Если долгов больше max_debt, возвращает true (студент — должник), иначе false.
}

void print(vector<Student> students) { // Функция выводит список студентов в определённом формате.
    if (students.empty()) { // Если список пуст — выводит сообщение. Иначе — печатает список.
        cout << "Empty list!";
    } else {
        for (size_t i = 0; i < students.size(); i++) { // Цикл по всем студентам.
            if (i == 0 || students[i].group != students[i-1].group) { // Если это первый студент (i == 0) или его группа отличается от предыдущего,
                cout << students[i].group << endl; // Тогда печатаем название группы.
            }
            cout << "- " << students[i].name << endl; // Печатаем имя студента с отступом - .
        }
    }
}

int main() // Объявляем главную функцию программы, с которой начинается выполнение.
{
    int cur_semester, max_debt, stud_count; // Считывает три числа:
											// stud_count — сколько студентов будет.
											// cur_semester — текущий семестр.
											// max_debt — максимальное число долгов.
	
    std::cin >> stud_count >> cur_semester >> max_debt; // stud_count — сколько студентов будет. cur_semester — текущий семестр. max_debt — максимальное число долгов.
    std::cin.ignore(1);  // Убираем из потока символ \n для корректной работы getline

    std::vector<Student> students(stud_count); // Создаём вектор студентов размером stud_count.
    for(auto& student: students) { // Цикл по каждому студенту:
        std::string line; // Объявляем переменную line
        std::getline(std::cin, line); // getline читает целую строку (включая пробелы, до \n).
        student = make_student(line); // make_student(line) превращает строку в объект Student.
    }
    
    std::vector<Student> expulsion_candidates; // Создаём пустой вектор для студентов, которых могут отчислить.
    std::copy_if(students.begin(), // Начинаем вызов функции std::copy_if — это готовая функция из библиотеки <algorithm>, которая копирует элементы из одного места в другое, но только если выполняется определённое условие.
                 students.end(), // Указываем конец диапазона, на который действует copy_if.
                 std::back_inserter(expulsion_candidates), // Указываем, куда копировать подходящих студентов.
                 [cur_semester, max_debt](const Student& student){ // Начинается лямбда-функция — это анонимная функция, которая создаётся прямо в коде и передаётся как условие.
				 												   // [cur_semester, max_debt] — захват переменных из внешнего контекста.
				 												   // (const Student& student) — параметр лямбды: она получает ссылку на студента (чтобы не копировать).

                     return is_debtor(student, cur_semester, max_debt); // Тело лямбды — вызов функции is_debtor, которая проверяет, является ли студент должником.
					 // Проверяет: сколько у студента несданных курсов, которые должны были быть сданы до текущего семестра.
					 // Если долгов больше max_debt — возвращает true, иначе false.
					 // copy_if копирует студента только если функция вернула true.
					 
                 });

    std::sort(expulsion_candidates.begin(), expulsion_candidates.end(), is_upper);
    print(expulsion_candidates);
}
```

```cpp
using json = nlohmann::json;
using namespace std;

int main() {
	ifstream file("data.json");
	json data = json::parse(file);
	int user_id, count = 0;
	cin >> user_id;
    for (auto& project : data) {
        for (auto& task : project["tasks"]) {
            if (task["user_id"] == user_id && task["completed"]) {
                count++;
            }
        }
    }
    cout << count;
}
```

```go
package main
import (
	"encoding/json"
	"fmt"
	"os"
)

type Task struct {
	UserID    int    `json:"user_id"`
	ID        int    `json:"id"`
	Title     string `json:"title"`
	Completed bool   `json:"completed"`
}

type Project struct {
	ProjectID int    `json:"project_id"`
	Tasks     []Task `json:"tasks"`
}

func main(){
	file, _ := os.Open("data.json")
	var projects []Project
	json.NewDecoder(file).Decode(&projects)
	var userID int
	fmt.Scan(&userID)
	count := 0
	for _, project := range projects {
	    for _, task := range project.Tasks {
	        if task.UserID == userID && task.Completed {
	            count++
	        }
	    }
	}
	fmt.Print(count)
}
```
